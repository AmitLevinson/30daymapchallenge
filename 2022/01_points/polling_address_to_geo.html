def geocode_address(loc):
    geocode_result = gmaps.geocode(loc + ', ×¨×—×•×‘×•×ª')
    if not geocode_result:
        location = {
        'address': loc,
        'lat' : np.NaN,
        'lon': np.NaN
    }
    else:
        lat = geocode_result[0]["geometry"]["location"]["lat"]
        lon = geocode_result[0]["geometry"]["location"]["lng"]
        location = {
            'address': loc,
            'lat' : lat,
            'lon': lon
        }
    df = pd.DataFrame([location])
    return df
